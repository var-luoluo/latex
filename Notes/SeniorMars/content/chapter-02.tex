\section{数列极限概念}
\dfn{数列极限}{limit}{%
  设 $a_n$ 为数列, $a$ 为定数. 对于任给的正数 $\epsilon$, 总存在正整数 $N$, 使得当 $n>N$ 时, 有
  \[
    |a_n - a| < \epsilon,
  \]
  则称数列 ${a_n}$ 收敛于 $a$, 并记作
  \[
    \lim_{n\to\infty} a_n = a, \ \text{或}\  a_n \to a(n\to\infty).
  \]
}

\nt{若数列 ${a_n}$ 没有极限，则称 ${a_n}$ 不收敛，或称 ${a_n}$ 为发散序列. 参见 \ref{def:limit}}
\ex{}{
  证明 $\lim\limits_{n\to\infty} \frac{1}{n^\alpha}=0$, 这里 $\alpha$ 是正数.\\
  \\
  由于
    \[
      \left|\frac{1}{n^\alpha}-0\right|=\frac{1}{n^\alpha},
    \]
    对任给的 $\epsilon > 0$, 只要取 $N=\left[\frac{1}{\epsilon^{\frac{1}{\alpha}}}\right]+1$, 则当 $n>N$ 时, 便有
    \[
    \frac{1}{n^\alpha}<\frac{1}{N^\alpha}<\epsilon, \ \text{即} \left|\frac{1}{n^\alpha}-0\right|<\epsilon.
    \]
    这就证明了 $\lim\limits_{n\to\infty} \frac{1}{n^\alpha}=0$.
}
